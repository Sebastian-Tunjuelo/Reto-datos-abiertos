## Checklist de criterios de aceptación

### Estructura de archivos
- [ ] `data/eva_historica.parquet` existe
- [ ] `data/eva_reciente.parquet` existe
- [ ] `data/eva_completa.parquet` existe

### Integridad de la unión
- [ ] `len(completa) == len(historica) + len(reciente)` exactamente
- [ ] `historica["año"].max() <= 2018`
- [ ] `reciente["año"].min() >= 2019`

### Esquema y tipos
- [ ] Las 12 columnas del esquema unificado están presentes en los 3 archivos
- [ ] `codigo_dane` es `object` (string) con longitud exacta de 5 en todos los registros
- [ ] `año` es `int64`
- [ ] `rendimiento` es `float64`
- [ ] `area_sembrada`, `area_cosechada`, `produccion` son `float64`
- [ ] `fuente` en histórica es siempre `'historica'`
- [ ] `fuente` en reciente es siempre `'reciente'`

### Valores de dominio
- [ ] `cultivo` contiene solo `{'Café', 'Cacao', 'Maíz'}` — sin otros valores
- [ ] `ciclo` contiene solo `{'PERMANENTE', 'TRANSITORIO', None/NaN}`
- [ ] No hay filas con `rendimiento == 0.0` (deben ser `NaN`)
- [ ] `rendimiento` está dentro de rangos por cultivo (o es `NaN`):
  - Café: 0.3–3.0 t/ha
  - Cacao: 0.2–2.5 t/ha
  - Maíz: 0.5–10.0 t/ha

### Cobertura del MVP
- [ ] Los 15 municipios del MVP están presentes en `eva_completa`
- [ ] `codigo_dane` tiene exactamente 15 valores únicos en `eva_completa`

### Duplicados
- [ ] No hay duplicados por `(codigo_dane, cultivo, año, periodo, fuente)` en ningún archivo

## Script de validación
Guardar como `specs/D1_pipeline_eva/validate_d1.py`:

```python
"""
Script de validación para el pipeline D1 — EVA.
Ejecutar después de modules.agricultural.ingestion.run_pipeline()
"""
import sys
from pathlib import Path

import pandas as pd

# Ajustar path para importar desde la raíz del proyecto
ROOT = Path(__file__).parent.parent.parent
sys.path.insert(0, str(ROOT))

from shared.config import DATA_DIR, RENDIMIENTO_RANGOS
from shared.dane_codes import MVP_CODIGOS

CULTIVOS_ESPERADOS = {"Café", "Cacao", "Maíz"}
CICLOS_VALIDOS = {"PERMANENTE", "TRANSITORIO"}


def run_validations(data_dir: Path = DATA_DIR) -> None:
    """Ejecuta todas las validaciones. Lanza AssertionError si alguna falla."""
    errores = []

    # --- Cargar archivos ---
    for nombre in ["eva_historica", "eva_reciente", "eva_completa"]:
        path = data_dir / f"{nombre}.parquet"
        assert path.exists(), f"❌ Archivo no encontrado: {path}"

    hist = pd.read_parquet(data_dir / "eva_historica.parquet")
    rec  = pd.read_parquet(data_dir / "eva_reciente.parquet")
    comp = pd.read_parquet(data_dir / "eva_completa.parquet")

    # --- Integridad de la unión ---
    assert len(comp) == len(hist) + len(rec), (
        f"❌ Integridad: completa({len(comp)}) != hist({len(hist)}) + rec({len(rec)})"
    )
    assert hist["año"].max() <= 2018, f"❌ Histórica tiene años > 2018: {hist['año'].max()}"
    assert rec["año"].min() >= 2019, f"❌ Reciente tiene años < 2019: {rec['año'].min()}"

    # --- Esquema ---
    columnas_esperadas = [
        "codigo_dane", "municipio", "departamento", "cultivo",
        "año", "periodo", "rendimiento", "area_sembrada",
        "area_cosechada", "produccion", "ciclo", "fuente"
    ]
    for df, nombre in [(hist, "historica"), (rec, "reciente"), (comp, "completa")]:
        faltantes = set(columnas_esperadas) - set(df.columns)
        assert not faltantes, f"❌ {nombre}: columnas faltantes: {faltantes}"

    # --- Tipos ---
    assert comp["codigo_dane"].str.len().eq(5).all(), "❌ codigo_dane con longitud != 5"
    assert comp["año"].dtype == "int64", f"❌ año no es int64: {comp['año'].dtype}"
    assert comp["rendimiento"].dtype == "float64", f"❌ rendimiento no es float64"

    # --- Valores de dominio ---
    cultivos_reales = set(comp["cultivo"].dropna().unique())
    assert cultivos_reales == CULTIVOS_ESPERADOS, (
        f"❌ Cultivos inesperados: {cultivos_reales - CULTIVOS_ESPERADOS}"
    )

    ciclos_reales = set(comp["ciclo"].dropna().unique())
    invalidos = ciclos_reales - CICLOS_VALIDOS
    assert not invalidos, f"❌ Valores de ciclo inválidos: {invalidos}"

    assert not (comp["rendimiento"] == 0.0).any(), "❌ Hay rendimientos == 0.0 (deben ser NaN)"

    for cultivo, (rmin, rmax) in RENDIMIENTO_RANGOS.items():
        sub = comp[comp["cultivo"] == cultivo]["rendimiento"].dropna()
        fuera = sub[(sub < rmin) | (sub > rmax)]
        assert fuera.empty, (
            f"❌ {cultivo}: {len(fuera)} rendimientos fuera de rango [{rmin}, {rmax}]"
        )

    # --- Cobertura MVP ---
    codigos_presentes = set(comp["codigo_dane"].unique())
    faltantes_mvp = set(MVP_CODIGOS) - codigos_presentes
    assert not faltantes_mvp, f"❌ Municipios MVP sin datos: {faltantes_mvp}"
    assert comp["codigo_dane"].nunique() == 15, (
        f"❌ Se esperaban 15 municipios, hay {comp['codigo_dane'].nunique()}"
    )

    # --- Fuentes ---
    assert (hist["fuente"] == "historica").all(), "❌ Histórica tiene fuente != 'historica'"
    assert (rec["fuente"] == "reciente").all(), "❌ Reciente tiene fuente != 'reciente'"

    # --- Duplicados ---
    llave = ["codigo_dane", "cultivo", "año", "periodo", "fuente"]
    for df, nombre in [(hist, "historica"), (rec, "reciente"), (comp, "completa")]:
        dups = df.duplicated(llave).sum()
        assert dups == 0, f"❌ {nombre}: {dups} duplicados por {llave}"

    # --- Resumen si todo pasó ---
    print("✅ Todos los criterios de aceptación pasaron")
    print(f"\nResumen:")
    print(f"  EVA histórica:  {len(hist):>6,} registros | años {hist['año'].min()}–{hist['año'].max()}")
    print(f"  EVA reciente:   {len(rec):>6,} registros | años {rec['año'].min()}–{rec['año'].max()}")
    print(f"  EVA completa:   {len(comp):>6,} registros | años {comp['año'].min()}–{comp['año'].max()}")
    print(f"\n  Municipios: {comp['codigo_dane'].nunique()}/15")
    print(f"  Registros con rendimiento=NaN: {comp['rendimiento'].isna().sum()} "
          f"({comp['rendimiento'].isna().mean():.1%})")
    for cultivo in ["Café", "Cacao", "Maíz"]:
        n = (comp["cultivo"] == cultivo).sum()
        med = comp[comp["cultivo"] == cultivo]["rendimiento"].mean()
        print(f"  {cultivo}: {n:>5} registros | rendimiento medio: {med:.2f} t/ha")


if __name__ == "__main__":
    run_validations()
```
