## Agente

**Rol**: Eres un ingeniero de QA de datos. Tu trabajo es escribir un script de validación independiente que actúe como árbitro objetivo del pipeline climático. No tienes acceso al código de producción — solo lees `clima_agregado.parquet` y verificas que cumple el contrato de datos definido en el README de D2.

**Mentalidad**: Piensa como un auditor, no como el autor del pipeline. Cada aserción debe ser independiente, tener un mensaje de error que explique exactamente qué falló y por qué importa. Prefiere aserciones granulares sobre aserciones genéricas. Un test que pasa cuando no debería es peor que no tener test.

**Restricciones**:

- No importar ni llamar funciones de `modules/climate/` — solo leer el Parquet
- No modificar el archivo Parquet bajo ninguna circunstancia
- Cada aserción debe tener un mensaje de error con el prefijo `❌` y descripción del problema
- Si una aserción falla, el script debe detenerse inmediatamente con `AssertionError`
- El script debe poder ejecutarse de forma aislada sin depender del estado del proceso anterior

## Script de validación

Guardar como `specs/D2_pipeline_clima/validate_d2.py`:

```python
"""
Script de validación para el pipeline D2 — Clima.
Ejecutar después de modules.climate.ingestion.run_pipeline()
"""
import sys
from pathlib import Path

import pandas as pd

ROOT = Path(__file__).parent.parent.parent
sys.path.insert(0, str(ROOT))

from shared.config import DATA_DIR
from shared.dane_codes import MVP_CODIGOS, DANE_TO_NAME

COLUMNAS_ESPERADAS = [
    "codigo_dane", "municipio", "año",
    "prec_acum_mm", "prec_dias_secos", "prec_dias_lluvia",
    "temp_media_c", "temp_max_media_c", "hum_media_pct",
    "n_estaciones_prec", "n_estaciones_temp",
    "anomalia_prec", "anomalia_temp",
]

AÑOS_ESPERADOS = list(range(2007, 2025))  # 2007–2024 inclusive
N_MUNICIPIOS = 15
N_AÑOS = 18
N_FILAS_ESPERADAS = N_MUNICIPIOS * N_AÑOS  # 270


def run_validations(data_dir: Path = DATA_DIR) -> None:
    """Ejecuta todas las validaciones. Lanza AssertionError si alguna falla."""

    # --- Archivo existe ---
    path = data_dir / "clima_agregado.parquet"
    assert path.exists(), f"❌ Archivo no encontrado: {path}"

    df = pd.read_parquet(path)

    # --- Dimensiones ---
    assert len(df) == N_FILAS_ESPERADAS, (
        f"❌ Se esperaban {N_FILAS_ESPERADAS} filas, hay {len(df)}"
    )
    assert len(df.columns) == len(COLUMNAS_ESPERADAS), (
        f"❌ Se esperaban {len(COLUMNAS_ESPERADAS)} columnas, hay {len(df.columns)}"
    )

    # --- Esquema: columnas presentes y en orden ---
    assert list(df.columns) == COLUMNAS_ESPERADAS, (
        f"❌ Columnas incorrectas o en orden incorrecto.\n"
        f"   Esperadas: {COLUMNAS_ESPERADAS}\n"
        f"   Reales:    {list(df.columns)}"
    )

    # --- Tipos ---
    assert df["codigo_dane"].str.len().eq(5).all(), (
        "❌ codigo_dane con longitud != 5"
    )
    assert df["año"].dtype == "int64", (
        f"❌ año no es int64: {df['año'].dtype}"
    )
    assert df["prec_acum_mm"].dtype == "float64", (
        f"❌ prec_acum_mm no es float64: {df['prec_acum_mm'].dtype}"
    )
    assert df["temp_media_c"].dtype == "float64", (
        f"❌ temp_media_c no es float64: {df['temp_media_c'].dtype}"
    )
    assert df["hum_media_pct"].dtype == "float64", (
        f"❌ hum_media_pct no es float64: {df['hum_media_pct'].dtype}"
    )
    assert df["n_estaciones_prec"].dtype == "int64", (
        f"❌ n_estaciones_prec no es int64: {df['n_estaciones_prec'].dtype}"
    )
    assert df["n_estaciones_temp"].dtype == "int64", (
        f"❌ n_estaciones_temp no es int64: {df['n_estaciones_temp'].dtype}"
    )

    # --- Rango de años ---
    años_reales = sorted(df["año"].unique().tolist())
    assert años_reales == AÑOS_ESPERADOS, (
        f"❌ Años incorrectos.\n"
        f"   Esperados: {AÑOS_ESPERADOS}\n"
        f"   Reales:    {años_reales}"
    )

    # --- Valores físicamente imposibles ---
    prec_neg = (df["prec_acum_mm"] < 0).sum()
    assert prec_neg == 0, f"❌ {prec_neg} valores negativos en prec_acum_mm"

    temp_invalida = df["temp_media_c"].dropna()
    temp_invalida = temp_invalida[(temp_invalida < 0) | (temp_invalida > 45)]
    assert temp_invalida.empty, (
        f"❌ {len(temp_invalida)} valores de temp_media_c fuera de 0–45°C"
    )

    hum_invalida = df["hum_media_pct"].dropna()
    hum_invalida = hum_invalida[(hum_invalida < 0) | (hum_invalida > 100)]
    assert hum_invalida.empty, (
        f"❌ {len(hum_invalida)} valores de hum_media_pct fuera de 0–100%"
    )

    # --- n_estaciones sin NaN y >= 0 ---
    assert df["n_estaciones_prec"].isna().sum() == 0, (
        "❌ n_estaciones_prec tiene NaN"
    )
    assert df["n_estaciones_temp"].isna().sum() == 0, (
        "❌ n_estaciones_temp tiene NaN"
    )
    assert (df["n_estaciones_prec"] >= 0).all(), (
        "❌ n_estaciones_prec tiene valores negativos"
    )
    assert (df["n_estaciones_temp"] >= 0).all(), (
        "❌ n_estaciones_temp tiene valores negativos"
    )

    # --- Cobertura del MVP ---
    codigos_presentes = set(df["codigo_dane"].unique())
    faltantes = set(MVP_CODIGOS) - codigos_presentes
    assert not faltantes, f"❌ Municipios MVP sin datos: {faltantes}"
    assert df["codigo_dane"].nunique() == N_MUNICIPIOS, (
        f"❌ Se esperaban {N_MUNICIPIOS} municipios, hay {df['codigo_dane'].nunique()}"
    )

    # --- Cada municipio tiene exactamente N_AÑOS filas ---
    filas_por_municipio = df.groupby("codigo_dane").size()
    mal_conteo = filas_por_municipio[filas_por_municipio != N_AÑOS]
    assert mal_conteo.empty, (
        f"❌ Municipios con número incorrecto de filas:\n{mal_conteo}"
    )

    # --- Integridad: municipio consistente con codigo_dane ---
    df_check = df[["codigo_dane", "municipio"]].drop_duplicates()
    for _, row in df_check.iterrows():
        esperado = DANE_TO_NAME.get(row["codigo_dane"])
        assert row["municipio"] == esperado, (
            f"❌ municipio inconsistente: codigo_dane={row['codigo_dane']} "
            f"tiene '{row['municipio']}', se esperaba '{esperado}'"
        )

    # --- Duplicados ---
    dups = df.duplicated(["codigo_dane", "año"]).sum()
    assert dups == 0, f"❌ {dups} duplicados por (codigo_dane, año)"

    # --- Cobertura de datos (advertencias) ---
    print("\n⚠️  Advertencias de cobertura (no bloquean el pipeline):")
    for variable in ["prec_acum_mm", "temp_media_c", "anomalia_prec", "anomalia_temp"]:
        cobertura = (
            df.groupby("codigo_dane")[variable]
            .apply(lambda s: s.notna().sum())
        )
        municipios_ok = (cobertura >= 10).sum()
        if municipios_ok < 10:
            print(
                f"   ⚠️  {variable}: solo {municipios_ok}/15 municipios "
                f"tienen ≥ 10 años con datos"
            )
        else:
            print(f"   ✅ {variable}: {municipios_ok}/15 municipios con ≥ 10 años de datos")

    # --- Resumen si todo pasó ---
    print("\n✅ Todos los criterios de aceptación pasaron")
    print(f"\nResumen:")
    print(f"  Filas totales:    {len(df):>6,} ({N_MUNICIPIOS} municipios × {N_AÑOS} años)")
    print(f"  Años cubiertos:   {df['año'].min()}–{df['año'].max()}")
    print(f"  Municipios:       {df['codigo_dane'].nunique()}/{N_MUNICIPIOS}")

    for col, label in [
        ("prec_acum_mm",  "Precipitación"),
        ("temp_media_c",  "Temperatura  "),
        ("hum_media_pct", "Humedad      "),
        ("anomalia_prec", "Anomalía prec"),
        ("anomalia_temp", "Anomalía temp"),
    ]:
        pct_nulo = df[col].isna().mean()
        media = df[col].mean()
        print(
            f"  {label}: media={media:>7.2f} | NaN={pct_nulo:.1%}"
        )


if __name__ == "__main__":
    run_validations()
```

| Advertencias de cobertura     | No aplica                               | Sí — cobertura por variable y municipio               |
| Consistencia municipio/código | No aplica                               | Cruzar `municipio` con `DANE_TO_NAME`                 |
