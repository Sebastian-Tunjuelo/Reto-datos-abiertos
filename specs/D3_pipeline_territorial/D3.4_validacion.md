## Agente

**Rol**: Eres un ingeniero de QA de datos. Tu trabajo es escribir un script de validación independiente que actúe como árbitro objetivo del pipeline territorial. No tienes acceso al código de producción — solo lees los archivos Parquet resultantes y verificas que cumplen el contrato de datos.

**Mentalidad**: Piensa como un auditor, no como el autor del pipeline. Los datos UPRA son estáticos, por lo que los errores aquí no son de cobertura temporal sino de integridad estructural: ¿están los 15 municipios? ¿suman los porcentajes 1.0? ¿hay duplicados? Cada aserción debe tener un mensaje de error que explique exactamente qué falló. Prefiere aserciones granulares sobre aserciones genéricas.

**Restricciones**:
- No importar ni llamar funciones del módulo `modules/territorial/` — solo leer Parquets
- No modificar los archivos Parquet bajo ninguna circunstancia
- Cada aserción debe tener un mensaje de error con el prefijo `❌` y descripción del problema
- El script debe poder ejecutarse de forma aislada sin depender del estado del proceso anterior
- Si una aserción falla, el script debe detenerse inmediatamente con `AssertionError`

## Script de validación
Guardar como `specs/D3_pipeline_territorial/validate_d3.py`:

```python
"""
Script de validación para el pipeline D3 — Territorial (UPRA).
Ejecutar después de modules.territorial.ingestion.run_pipeline()
"""
import sys
from pathlib import Path

import pandas as pd
import numpy as np

ROOT = Path(__file__).parent.parent.parent
sys.path.insert(0, str(ROOT))

from shared.config import DATA_DIR
from shared.dane_codes import MVP_CODIGOS

COLUMNAS_APTITUD = [
    "codigo_dane", "municipio", "departamento", "cultivo",
    "area_alta_ha", "area_media_ha", "area_baja_ha", "area_exclusion_ha",
    "area_total_ha", "pct_alta", "pct_media", "pct_baja", "pct_exclusion",
]
COLUMNAS_FRONTERA = [
    "codigo_dane", "municipio", "departamento",
    "area_condicionada_ha", "area_no_condicionada_ha",
    "area_total_ha", "pct_condicionada", "pct_no_condicionada",
]
TOLERANCIA_PCT = 0.001


def run_validations(data_dir: Path = DATA_DIR) -> None:
    """Ejecuta todas las validaciones. Lanza AssertionError si alguna falla."""

    # --- Existencia de archivos ---
    for nombre in ["aptitud_cafe", "aptitud_cacao", "aptitud_maiz", "frontera"]:
        path = data_dir / f"{nombre}.parquet"
        assert path.exists(), f"❌ Archivo no encontrado: {path}"

    cafe  = pd.read_parquet(data_dir / "aptitud_cafe.parquet")
    cacao = pd.read_parquet(data_dir / "aptitud_cacao.parquet")
    maiz  = pd.read_parquet(data_dir / "aptitud_maiz.parquet")
    front = pd.read_parquet(data_dir / "frontera.parquet")

    # --- Esquema de aptitud ---
    for df, nombre in [(cafe, "aptitud_cafe"), (cacao, "aptitud_cacao"), (maiz, "aptitud_maiz")]:
        faltantes = set(COLUMNAS_APTITUD) - set(df.columns)
        assert not faltantes, f"❌ {nombre}: columnas faltantes: {faltantes}"

    # --- Esquema de frontera ---
    faltantes_front = set(COLUMNAS_FRONTERA) - set(front.columns)
    assert not faltantes_front, f"❌ frontera: columnas faltantes: {faltantes_front}"

    # --- Tipos y valores de codigo_dane ---
    for df, nombre in [(cafe, "cafe"), (cacao, "cacao"), (maiz, "maiz"), (front, "frontera")]:
        assert df["codigo_dane"].str.len().eq(5).all(), (
            f"❌ {nombre}: codigo_dane con longitud != 5"
        )

    # --- Columna cultivo ---
    assert (cafe["cultivo"] == "Café").all(),  "❌ aptitud_cafe: cultivo != 'Café'"
    assert (cacao["cultivo"] == "Cacao").all(), "❌ aptitud_cacao: cultivo != 'Cacao'"
    assert (maiz["cultivo"] == "Maíz").all(),  "❌ aptitud_maiz: cultivo != 'Maíz'"

    # --- Tipos numéricos ---
    cols_float_aptitud = [
        "area_alta_ha", "area_media_ha", "area_baja_ha", "area_exclusion_ha",
        "area_total_ha", "pct_alta", "pct_media", "pct_baja", "pct_exclusion",
    ]
    for df, nombre in [(cafe, "cafe"), (cacao, "cacao"), (maiz, "maiz")]:
        for col in cols_float_aptitud:
            assert df[col].dtype == "float64", (
                f"❌ {nombre}.{col} no es float64: {df[col].dtype}"
            )

    cols_float_front = [
        "area_condicionada_ha", "area_no_condicionada_ha",
        "area_total_ha", "pct_condicionada", "pct_no_condicionada",
    ]
    for col in cols_float_front:
        assert front[col].dtype == "float64", (
            f"❌ frontera.{col} no es float64: {front[col].dtype}"
        )

    # --- area_total_ha > 0 ---
    for df, nombre in [(cafe, "cafe"), (cacao, "cacao"), (maiz, "maiz"), (front, "frontera")]:
        assert (df["area_total_ha"] > 0).all(), (
            f"❌ {nombre}: hay registros con area_total_ha <= 0"
        )

    # --- Rango de porcentajes [0, 1] ---
    for df, nombre in [(cafe, "cafe"), (cacao, "cacao"), (maiz, "maiz")]:
        for col in ["pct_alta", "pct_media", "pct_baja", "pct_exclusion"]:
            fuera = df[(df[col] < 0) | (df[col] > 1)]
            assert fuera.empty, f"❌ {nombre}.{col}: valores fuera de [0, 1]: {fuera['codigo_dane'].tolist()}"

    for col in ["pct_condicionada", "pct_no_condicionada"]:
        fuera = front[(front[col] < 0) | (front[col] > 1)]
        assert fuera.empty, f"❌ frontera.{col}: valores fuera de [0, 1]"

    # --- Suma de porcentajes ≈ 1.0 ---
    for df, nombre in [(cafe, "cafe"), (cacao, "cacao"), (maiz, "maiz")]:
        suma = df[["pct_alta", "pct_media", "pct_baja", "pct_exclusion"]].sum(axis=1)
        fuera = df[(suma - 1.0).abs() > TOLERANCIA_PCT]
        assert fuera.empty, (
            f"❌ {nombre}: porcentajes no suman 1.0 en municipios: {fuera['codigo_dane'].tolist()}"
        )

    suma_front = front[["pct_condicionada", "pct_no_condicionada"]].sum(axis=1)
    fuera_front = front[(suma_front - 1.0).abs() > TOLERANCIA_PCT]
    assert fuera_front.empty, (
        f"❌ frontera: porcentajes no suman 1.0 en municipios: {fuera_front['codigo_dane'].tolist()}"
    )

    # --- Cobertura MVP ---
    for df, nombre in [(cafe, "cafe"), (cacao, "cacao"), (maiz, "maiz"), (front, "frontera")]:
        presentes = set(df["codigo_dane"].unique())
        faltantes_mvp = set(MVP_CODIGOS) - presentes
        assert not faltantes_mvp, f"❌ {nombre}: municipios MVP sin datos: {faltantes_mvp}"
        assert df["codigo_dane"].nunique() == 15, (
            f"❌ {nombre}: se esperaban 15 municipios, hay {df['codigo_dane'].nunique()}"
        )

    # --- Duplicados ---
    for df, nombre in [(cafe, "cafe"), (cacao, "cacao"), (maiz, "maiz"), (front, "frontera")]:
        dups = df.duplicated("codigo_dane").sum()
        assert dups == 0, f"❌ {nombre}: {dups} duplicados por codigo_dane"

    # --- Resumen si todo pasó ---
    print("✅ Todos los criterios de aceptación pasaron")
    print("\nResumen:")
    for df, nombre, cultivo_col in [
        (cafe,  "aptitud_cafe",  "cultivo"),
        (cacao, "aptitud_cacao", "cultivo"),
        (maiz,  "aptitud_maiz",  "cultivo"),
    ]:
        print(f"\n  {nombre}: {len(df)} municipios | cultivo: {df[cultivo_col].iloc[0]}")
        print(f"    pct_alta  media: {df['pct_alta'].mean():.1%}")
        print(f"    pct_media media: {df['pct_media'].mean():.1%}")
        print(f"    pct_baja  media: {df['pct_baja'].mean():.1%}")
        print(f"    pct_excl  media: {df['pct_exclusion'].mean():.1%}")

    print(f"\n  frontera: {len(front)} municipios")
    print(f"    pct_condicionada     media: {front['pct_condicionada'].mean():.1%}")
    print(f"    pct_no_condicionada  media: {front['pct_no_condicionada'].mean():.1%}")


if __name__ == "__main__":
    run_validations()
```
